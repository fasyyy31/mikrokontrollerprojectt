void handleRoot() { server.send(200, "text/html", index_html); }

 

void handleStatus() {

  String json = "{";

  json += "\"suhu\":\"" + String(suhuSekarang, 1) + "\",";

  json += "\"katup\":\"" + statusKatup + "\",";

  json += "\"mode\":" + String(modeManual);

  json += "}";

  server.send(200, "application/json", json);

}

 

void handleControl() {

  if (server.hasArg("aksi")) {

    String aksi = server.arg("aksi");

    if (aksi == "buka") {

      modeManual = true;

      digitalWrite(pinRelay, LOW);

      statusKatup = "TERBUKA";

    } else if (aksi == "tutup") {

      modeManual = true;

      digitalWrite(pinRelay, HIGH);

      statusKatup = "TERTUTUP";

    } else if (aksi == "auto") {

      modeManual = false;

    }

  }

  server.send(200, "text/plain", "OK");

}

 

void setup() {

  Serial.begin(9600);

  delay(500);

 

  if (!thermocouple.begin()) {

    Serial.println("ERROR: Sensor MAX31855 bermasalah!");

  }

  

  pinMode(pinRelay, OUTPUT);

  digitalWrite(pinRelay, HIGH);

 

  WiFi.softAP(ssid, password);

  Serial.print("IP Address Web: "); Serial.println(WiFi.softAPIP());

  

  server.on("/", handleRoot);

  server.on("/status", handleStatus);

  server.on("/kontrol", handleControl);

  server.begin();

}

 

void loop() {

  server.handleClient();

  double c = thermocouple.readCelsius();

  if (!isnan(c)) { suhuSekarang = c; }

 

  if (!modeManual) {

    if (suhuSekarang >= SUHU_TARGET) {

      digitalWrite(pinRelay, LOW);

      statusKatup = "TERBUKA";

    } else {

      digitalWrite(pinRelay, HIGH);

      statusKatup = "TERTUTUP";

    }

  }

  delay(200);

}
